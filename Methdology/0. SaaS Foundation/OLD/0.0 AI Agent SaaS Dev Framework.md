### 핵심 설계 원칙

| 원칙 | 설명 |
|------|------|
| Stateless API | 서버는 상태를 저장하지 않음 → 수평 확장 가능 |
| Multi-tenant First | 모든 리소스는 테넌트(조직) 단위로 격리 |
| API-First | 모든 기능은 API로 제공, UI는 API의 소비자 |
| Event-Driven | 도메인 간 결합은 이벤트/메시지 큐로 처리 |
| Observability-First | 모든 컴포넌트에 로깅·메트릭·트레이싱 내장 |

### 데이터 모델

```sql

  description      TEXT

  allowed_org_ids  JSONB  DEFAULT '[]'     -- 특정 조직 UUID 배열 (베타)

  updated_at       TIMESTAMPTZ
```

### 플래그 평가 로직

```python
# core/feature_flags.py
async def is_feature_enabled(
    flag_key: str,
    org_id: UUID,
    plan: str
) -> bool:
    flag = await get_flag(flag_key)
    if not flag or not flag.is_active:
        return False
    if flag.is_global:
        return True
    if plan in flag.allowed_plans:
        return True
    if str(org_id) in flag.allowed_org_ids:
        return True
    # 점진적 롤아웃: org_id 해시값 기반
    if flag.rollout_percent > 0:
        hash_val = int(hashlib.md5(str(org_id).encode()).hexdigest(), 16)
        return (hash_val % 100) < flag.rollout_percent
    return False
```

```python
from datetime import datetime, timezone, timedelta
import pytz

def now_utc() -> datetime:
    return datetime.now(timezone.utc)

def to_kst(dt: datetime) -> datetime:
    kst = pytz.timezone("Asia/Seoul")
    return dt.astimezone(kst)

def format_relative(dt: datetime) -> str:
    """'3분 전', '2시간 전' 형태로 반환"""
    diff = datetime.now(timezone.utc) - dt
    if diff.seconds < 60:
        return "방금 전"
    elif diff.seconds < 3600:
        return f"{diff.seconds // 60}분 전"
    elif diff.days < 1:
        return f"{diff.seconds // 3600}시간 전"
    else:
        return f"{diff.days}일 전"
```

```python
# app/utils/pagination.py
from dataclasses import dataclass
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, func

@dataclass
class PageResult:
    items: list
    total: int
    page: int
    size: int
    total_pages: int

async def paginate(session, query, page: int, size: int) -> PageResult:
    total = await session.scalar(select(func.count()).select_from(query.subquery()))
    items = await session.scalars(query.offset((page - 1) * size).limit(size))
    return PageResult(
        items=list(items),
        total=total,
        page=page,
        size=size,
        total_pages=(total + size - 1) // size
    )
```

## #09 Authentication Strategy

### 인증 흐름 (JWT + Refresh Rotate)

```
로그인 요청
    → [BE] 이메일/비밀번호 or OAuth 검증
    → [BE] Access Token (15분) + Refresh Token (30일) 발급
    → [FE] Access Token: 메모리 저장 (Zustand)
    → [FE] Refresh Token: HttpOnly Cookie (SameSite=Strict)

API 요청
    → [FE] Authorization: Bearer {access_token} 헤더 첨부
    → [BE] 미들웨어에서 JWT 검증 + tenant_id 추출

Access Token 만료
    → [FE] /auth/refresh 자동 호출 (Axios 인터셉터)
    → [BE] Refresh Token 검증 + 새 토큰 쌍 발급 (Rotate)
    → [BE] 이전 Refresh Token Redis Blacklist 등록

로그아웃
    → [FE] Access Token 메모리에서 제거
    → [BE] Refresh Token Redis Blacklist 등록
```

## #11 RBAC / ABAC

### RBAC (현재 적용)

역할 기반 접근 제어입니다. 구현이 단순하고 대부분의 B2B SaaS 요구사항을 충족합니다.

```python
# FastAPI 권한 의존성
def require_permission(*permissions: str):
    async def checker(
        current_user: UserContext = Depends(get_current_user)
    ):
        role_permissions = ROLE_PERMISSIONS[current_user.role]
        for perm in permissions:
            if perm not in role_permissions:
                raise ForbiddenException()
        return current_user
    return checker

# 사용 예시
@router.delete("/{agent_id}")
async def delete_agent(
    agent_id: UUID,
    user = Depends(require_permission("agent:manage"))
):
    ...
```

### ABAC (확장 계획)

속성 기반 접근 제어. 복잡한 정책이 필요할 때 단계적으로 도입합니다.

```python
# 예: 본인이 만든 Agent만 삭제 가능
async def can_delete_agent(user: UserContext, agent: Agent) -> bool:
    # RBAC: admin 이상
    if user.role in ["owner", "admin"]:
        return True
    # ABAC: 본인 소유
    if agent.created_by == user.user_id:
        return True
    return False
```

## #12 Session Management

### 세션 데이터 (Redis)

```python
# Redis key 구조
# access_token_blacklist:{jti}  → TTL = 토큰 남은 만료 시간
# refresh_token:{user_id}:{jti} → TTL = 30일
# login_attempts:{email}        → TTL = 15분

# 세션 관리 구현
class SessionManager:
    async def blacklist_token(self, jti: str, ttl: int):
        await redis.setex(f"access_token_blacklist:{jti}", ttl, "1")

    async def is_blacklisted(self, jti: str) -> bool:
        return await redis.exists(f"access_token_blacklist:{jti}") > 0

    async def increment_login_attempts(self, email: str) -> int:
        key = f"login_attempts:{email}"
        count = await redis.incr(key)
        if count == 1:
            await redis.expire(key, 900)  # 15분
        return count
```

## #16 Tenant Isolation Strategy

### 격리 방식: Pool (Row-level, organization_id)

모든 테이블에 `organization_id`를 포함하여, 애플리케이션 레벨에서 데이터를 격리합니다.

```python
# SQLAlchemy Event를 통한 자동 테넌트 필터
# db/session.py
from sqlalchemy import event
from contextvars import ContextVar

current_tenant_id: ContextVar[UUID | None] = ContextVar(
    "current_tenant_id", default=None
)

@event.listens_for(AsyncSession, "before_attach")
def set_tenant_filter(session, instance):
    tenant_id = current_tenant_id.get()
    if tenant_id and hasattr(instance, "organization_id"):
        # organization_id 자동 설정
        instance.organization_id = tenant_id
```

### 테넌트 컨텍스트 주입

```python
# api/deps.py
async def get_tenant_context(
    request: Request,
    token: str = Depends(oauth2_scheme)
) -> TenantContext:
    payload = decode_jwt(token)
    org_id = UUID(payload["org_id"])
    # Request State에 tenant_id 저장
    request.state.org_id = org_id
    current_tenant_id.set(org_id)
    return TenantContext(
        org_id=org_id,
        user_id=UUID(payload["sub"]),
        role=payload["role"],
        plan=payload["plan"]
    )
```

```python
# 민감 데이터 마스킹 유틸리티
def mask_email(email: str) -> str:
    local, domain = email.split("@")
    return f"{local[:2]}***@{domain}"
    # user@example.com → us***@example.com

def mask_phone(phone: str) -> str:
    return re.sub(r'(\d{3})\d{4}(\d{4})', r'\1-****-\2', phone)
    # 010-1234-5678 → 010-****-5678

def mask_card(number: str) -> str:
    return f"{'*' * 12}{number[-4:]}"
    # 1234567890123456 → ************3456
```

```python
class SoftDeleteMixin:

        DateTime(timezone=True), nullable=True, default=None
```

```python
# Webhook 발송 (HMAC 서명 포함)
async def deliver_webhook(webhook: Webhook, event_type: str, payload: dict):
    body = json.dumps({"event": event_type, "data": payload})
    signature = hmac.new(
        webhook.secret.encode(),
        body.encode(),
        hashlib.sha256
    ).hexdigest()
    headers = {
        "Content-Type": "application/json",
        "X-Webhook-Signature": f"sha256={signature}",
        "X-Webhook-Event": event_type,
    }
    # 재시도: 1분, 5분, 30분, 2시간, 5시간
    await httpx.post(webhook.url, content=body, headers=headers)
```

```python
from typing import TypedDict, Annotated
```

## #64 Agent Communication Protocol

### 스트리밍 응답 (SSE)

```python
# Server-Sent Events로 실시간 Agent 출력 스트리밍
@router.post("/{agent_id}/run/stream")
async def run_agent_stream(
    agent_id: UUID,
    request: AgentRunRequest,
    ctx: TenantContext = Depends(get_tenant_context),
):
    async def event_generator():
        async for chunk in execute_agent_stream(agent_id, request.input, ctx):
            if chunk.type == "token":
                yield f"data: {json.dumps({'type': 'token', 'content': chunk.text})}\n\n"
            elif chunk.type == "tool_start":
                yield f"data: {json.dumps({'type': 'tool_start', 'tool': chunk.tool_name})}\n\n"
            elif chunk.type == "tool_end":
                yield f"data: {json.dumps({'type': 'tool_end', 'result': chunk.result})}\n\n"
            elif chunk.type == "done":
                yield f"data: {json.dumps({'type': 'done', 'run_id': str(chunk.run_id)})}\n\n"

    return StreamingResponse(event_generator(), media_type="text/event-stream")
```

### Frontend 스트리밍 수신

```typescript
// hooks/useAgentStream.ts
export function useAgentStream(agentId: string) {
  const [output, setOutput] = useState('');
  const [status, setStatus] = useState<'idle' | 'running' | 'done' | 'error'>('idle');

  const run = async (input: string) => {
    setStatus('running');
    setOutput('');

    const response = await fetch(`/api/v1/agents/${agentId}/run/stream`, {
      method: 'POST',
      body: JSON.stringify({ input }),
      headers: { 'Content-Type': 'application/json' },
    });

    const reader = response.body!.getReader();
    const decoder = new TextDecoder();

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      const lines = decoder.decode(value).split('\n\n');
      for (const line of lines) {
        if (line.startsWith('data: ')) {
          const event = JSON.parse(line.slice(6));
          if (event.type === 'token') setOutput(prev => prev + event.content);
          if (event.type === 'done') setStatus('done');
        }
      }
    }
  };

  return { output, status, run };
}
```

```python
import openai

class InputGuardrail:
    async def validate(self, user_input: str, org_id: UUID) -> GuardrailResult:
        # 1. 길이 제한
        if len(user_input) > 50000:
            return GuardrailResult.block("입력이 너무 깁니다 (최대 50,000자).")

        # 2. OpenAI Moderation API
        moderation = await openai.moderations.create(input=user_input)
        if moderation.results[0].flagged:
            categories = moderation.results[0].categories
            raise AgentGuardrailException(
                f"유해 콘텐츠 감지: {[k for k,v in categories if v]}"
            )

        # 3. 프롬프트 인젝션 패턴 감지
        injection_patterns = [
            r"ignore (previous|above|all) instructions",
            r"you are now",
            r"jailbreak",
            r"DAN mode",
        ]
        for pattern in injection_patterns:
            if re.search(pattern, user_input, re.IGNORECASE):
                raise AgentGuardrailException("프롬프트 인젝션 시도 감지")

        return GuardrailResult.allow()

class OutputGuardrail:
    async def validate(self, output: str) -> str:
        # PII 마스킹
        output = mask_pii(output)
        # 유해 콘텐츠 재검증
        moderation = await openai.moderations.create(input=output)
        if moderation.results[0].flagged:
            return "죄송합니다. 이 요청에 대한 응답을 제공할 수 없습니다."
        return output
```
