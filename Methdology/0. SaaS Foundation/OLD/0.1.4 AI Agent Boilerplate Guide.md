```python
from sqlalchemy.orm import Mapped, mapped_column
from sqlalchemy.dialects.postgresql import UUID as PGUUID
from uuid import UUID, uuid4
from app.db.base import Base

class TenantBaseModel(Base):
    """모든 Tenant 격리 테이블의 부모. tenant_id 자동 포함."""

        PGUUID(as_uuid=True), nullable=False, index=True

    created_at: Mapped[DateTime] = mapped_column(
        DateTime(timezone=True), server_default=func.now()
    )
    updated_at: Mapped[DateTime] = mapped_column(
        DateTime(timezone=True), server_default=func.now(), onupdate=func.now()
    )
    created_by: Mapped[UUID | None] = mapped_column(PGUUID(as_uuid=True), nullable=True)
    is_active:  Mapped[bool] = mapped_column(Boolean, default=True)
    is_deleted: Mapped[bool] = mapped_column(Boolean, default=False)

class SoftDeleteMixin:
    is_active:  Mapped[bool]
    is_deleted: Mapped[bool]
```

```python
import redis.asyncio as redis
import json
from app.core.config import settings

redis_client: redis.Redis = None

async def get_redis() -> redis.Redis:
    global redis_client
    if redis_client is None:
        redis_client = redis.from_url(settings.redis_url)
    return redis_client

async def cache_get(key: str):
    r = await get_redis()
    value = await r.get(key)
    return json.loads(value) if value else None

async def cache_set(key: str, value, ttl: int = 300):
    r = await get_redis()
    await r.setex(key, ttl, json.dumps(value, default=str))

async def cache_delete(key: str):
    r = await get_redis()
    await r.delete(key)
```

#### Step 9 — `core/security.py`

```python
from passlib.context import CryptContext
from jose import JWTError, jwt
from datetime import datetime, timezone, timedelta
from uuid import UUID
import secrets
from app.core.config import settings

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto", bcrypt__rounds=12)

def hash_password(password: str) -> str:
    return pwd_context.hash(password)

def verify_password(plain: str, hashed: str) -> bool:
    return pwd_context.verify(plain, hashed)

def create_access_token(user_id: UUID, org_id: UUID, role: str) -> str:
    expire = datetime.now(timezone.utc) + timedelta(
        minutes=settings.access_token_expire_minutes
    )
    return jwt.encode(
        {"sub": str(user_id), "org_id": str(org_id), "role": role,
         "jti": secrets.token_urlsafe(16), "exp": expire, "type": "access"},
        settings.secret_key, algorithm="HS256"
    )

def require_role(*roles: str):
    """FastAPI 의존성 — 허용 Role 검증."""
    from fastapi import Depends, HTTPException
    def _check(token_data = Depends(verify_token)):
        if token_data.get("role") not in roles:
            raise HTTPException(status_code=403, detail="권한이 없습니다")
        return token_data
    return _check

def api_success(data=None, message: str = "") -> dict:
    return {"success": True, "data": data, "message": message}

def api_error(code: str, message: str, status_code: int = 400) -> dict:
    return {"success": False, "error": {"code": code, "message": message}}

def verify_token(token: str) -> dict:
    try:
        return jwt.decode(token, settings.secret_key, algorithms=["HS256"])
    except JWTError:
        raise UnauthorizedException("토큰이 유효하지 않습니다")
```

### ① base_agent.py

**경로:** `backend/src/core/base_agent.py`

```python
from __future__ import annotations
from abc import ABC, abstractmethod
from enum import Enum
from uuid import UUID, uuid4
from datetime import datetime, timezone
from typing import Any
from dataclasses import dataclass, field

class AuthorityLevel(str, Enum):
    L0 = "L0"   # 자동 실행 가능 (탐지·알림)
    L1 = "L1"   # Human 승인 필수
    L2 = "L2"   # 설계 승인 (Architect)
    L3 = "L3"   # Governance Board 승인

class RiskLevel(str, Enum):
    LOW      = "LOW"
    MEDIUM   = "MEDIUM"
    HIGH     = "HIGH"
    CRITICAL = "CRITICAL"

class MessageType(str, Enum):
    REQUEST      = "REQUEST"
    RESPONSE     = "RESPONSE"
    ESCALATION   = "ESCALATION"
    NOTIFICATION = "NOTIFICATION"

@dataclass
class AgentContext:
    trace_id:        UUID
    agent_code:      str
    user_id:         UUID
    context_version: str
    risk_level:      RiskLevel = RiskLevel.LOW
    payload:         dict      = field(default_factory=dict)
    parent_trace_id: UUID | None = None

@dataclass
class AgentResult:
    success:           bool
    data:              Any
    trace_id:          UUID
    risk_level:        RiskLevel = RiskLevel.LOW
    requires_approval: bool      = False
    message:           str       = ""

@dataclass
class AgentMessage:
    """Agent 간 통신 표준 메시지 — 12개 필드 모두 필수."""
    message_id:        UUID        = field(default_factory=uuid4)
    sender_agent:      str         = ""
    receiver_agent:    str         = ""       # agent_code 또는 'human'
    message_type:      MessageType = MessageType.REQUEST
    payload:           dict        = field(default_factory=dict)
    context_version:   str         = ""
    risk_level:        RiskLevel   = RiskLevel.LOW
    requires_approval: bool        = False
    created_at:        datetime    = field(
        default_factory=lambda: datetime.now(timezone.utc)
    )
    trace_id:          UUID        = field(default_factory=uuid4)
```

**경로:** `backend/src/agents/orchestrator_agent.py`

```python
from src.core.base_agent import BaseAgent, AgentContext, AgentResult, AuthorityLevel

class OrchestratorAgent(BaseAgent):
    name            = "Orch"
    authority_level = AuthorityLevel.L1
    allowed_tools   = ["api_call_internal", "send_notification"]
    _registry: dict[str, BaseAgent] = {}

    @classmethod
    def register(cls, pattern: str, agent: BaseAgent) -> None:
        """신규 Agent 완성 후 반드시 호출."""
        cls._registry[pattern] = agent

    async def execute(self, ctx: AgentContext, db=None) -> AgentResult:
        await self.load_global_context()
        target = self._route(ctx)
        if not target:
            return AgentResult(
                success=False, data={}, trace_id=ctx.trace_id,
                message=f"라우팅 실패: {ctx.payload.get('action', '')}"
            )
        if not await self._check_gate(ctx):
            return AgentResult(
                success=False, data={}, trace_id=ctx.trace_id,
                requires_approval=True, message="Human 승인 대기 중 (P1 Pre-approval)"
            )
        from uuid import uuid4
        sub_ctx = AgentContext(
            trace_id=ctx.trace_id, agent_code=target.name,
            user_id=ctx.user_id, context_version=ctx.context_version,
            risk_level=ctx.risk_level, payload=ctx.payload,
            parent_trace_id=ctx.trace_id,
        )
        return await target.run(sub_ctx, db)

    def _route(self, ctx: AgentContext) -> BaseAgent | None:
        action = ctx.payload.get("action", "")
        for pattern, agent in self._registry.items():
            if action.startswith(pattern.rstrip("*")):
                return agent
        return None

    async def _check_gate(self, ctx: AgentContext) -> bool:
        # Phase 3: approval_requests 테이블 조회 구현
        return ctx.payload.get("approved", False)
```

```typescript
import { useState, useCallback } from "react"
import { v4 as uuidv4 } from "uuid"
import type { AgentRequest, AgentResponse, ApprovalState } from "../types/governance"

export function useAgent() {
  const [isLoading,     setIsLoading]     = useState(false)
  const [error,         setError]         = useState<string | null>(null)
  const [approvalState, setApprovalState] = useState<ApprovalState | null>(null)

  const run = useCallback(async (req: AgentRequest): Promise<AgentResponse> => {
    setIsLoading(true); setError(null)
    const traceId = req.trace_id ?? uuidv4()
    try {
      const res = await fetch("/api/v1/agents/invoke", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ ...req, trace_id: traceId }),
      })
      const data: AgentResponse = await res.json()
      if (data.requires_approval) {
        setApprovalState({
          traceId, agentCode: req.agent_code, actionType: req.action,
          requestedAt: new Date().toISOString(), status: "PENDING",
        })
      }
      return data
    } catch (err) {
      const msg = err instanceof Error ? err.message : "Agent 호출 오류"
      setError(msg); throw err
    } finally {
      setIsLoading(false)
    }
  }, [])

  return { run, isLoading, error, approvalState, clearApproval: () => setApprovalState(null) }
}
```

```tsx
import { useState } from "react"
import type { ApprovalStatus, RiskLevel } from "../types/governance"

interface ApprovalGateProps {
  pattern:     "P1" | "P2" | "P3";  level:       "L1" | "L2" | "L3"
  traceId:     string;               agentCode:   string
  actionLabel: string;               riskLevel?:  RiskLevel
  onApprove:   (traceId: string) => void | Promise<void>
  onReject?:   (traceId: string, reason: string) => void | Promise<void>
  onRevise?:   (traceId: string, note: string)   => void | Promise<void>
  children?:   React.ReactNode
}

export function ApprovalGate({ pattern, level, traceId, agentCode,
  actionLabel, riskLevel = "LOW", onApprove, onReject, onRevise, children,
}: ApprovalGateProps) {
  const [status, setStatus]             = useState<ApprovalStatus>("PENDING")
  const [isProcessing, setIsProcessing] = useState(false)
  const [reason, setReason]             = useState("")

  const handleApprove = async () => {
    setIsProcessing(true); await onApprove(traceId)
    setStatus("APPROVED"); setIsProcessing(false)
  }
  const handleReject = async () => {
    if (!reason.trim()) return
    setIsProcessing(true); await onReject?.(traceId, reason)
    setStatus("REJECTED"); setIsProcessing(false)
  }

  if (pattern === "P1" && status === "PENDING") return (
    <div className="border-2 border-yellow-400 rounded-lg p-4 bg-yellow-50">
      <p className="font-bold text-yellow-600 mb-2">◆ 사전 승인 필요 [{level} · {riskLevel}]</p>

      <p className="text-xs text-gray-400 mb-3">trace_id: {traceId}</p>
      <div className="flex gap-2">
        <button onClick={handleApprove} disabled={isProcessing}
          className="px-4 py-2 bg-blue-600 text-white text-sm rounded disabled:opacity-50">
          {isProcessing ? "처리 중..." : "✅ 승인"}
        </button>
        <button onClick={handleReject} disabled={isProcessing || !reason.trim()}
          className="px-4 py-2 bg-red-500 text-white text-sm rounded disabled:opacity-50">
          ❌ 거부
        </button>
      </div>
      <input value={reason} onChange={e => setReason(e.target.value)}
        placeholder="거부 사유 (거부 시 필수)"
        className="mt-2 w-full border px-3 py-1.5 text-sm rounded" />
    </div>
  )
```

## 9. 참고사항 및 주의점

### 9.1 의존성 순서를 반드시 지켜야 하는 이유

Python에서 모듈을 import할 때 해당 모듈이 의존하는 다른 모듈이 먼저 존재해야 한다. `session.py`는 `config.py`의 `settings`를 import하므로, `config.py`가 먼저 완성되어 있어야 한다. 순서를 무시하면 `ImportError` 또는 순환 참조(Circular Import)가 발생한다.

### 9.2 PK는 반드시 UUID

`saas_foundation.json` 기준: `id UUID PRIMARY KEY DEFAULT gen_random_uuid()`. `Integer autoincrement`는 절대 사용하지 않는다.

### 9.3 환경변수 관리

- `.env` 파일은 절대 Git에 커밋하지 않는다.
- `.env.example`은 실제 값 없이 키만 포함하여 Git에 커밋한다.
- 새 환경변수 추가 시 `.env.example`도 동시에 업데이트한다.
- 코드 내 Secret 하드코딩 절대 금지 — `settings.secret_key` 경유.

### 9.4 SQLAlchemy async 사용 시

- `asyncpg` 드라이버 필수 — `DATABASE_URL`에 `postgresql+asyncpg://` 형식 사용.
- 모든 DB 쿼리는 `async def` + `await` 사용.
- 세션은 반드시 `Depends(get_db)` 의존성 주입 사용.

### 9.5 절대 금지 사항 요약

| 금지 행동 | 대신 사용 |
|----------|----------|
| `BaseAgent` 상속 없이 Agent 생성 | `BaseAgent` 상속 필수 |
| `anthropic.Anthropic().messages.create()` 직접 호출 | `LLMService.complete()` |
| Orch 경유 없이 Agent 직접 호출 | `OrchestratorAgent.run()` |
| `Base` 직접 상속 DB 모델 | `TenantBaseModel` 상속 |
| `Integer` PK | `UUID` PK |
| `db.delete(record)` 물리 삭제 | `.soft_delete()` |
| `fetch()` 직접 Agent 호출 | `useAgent().run()` |
| `governance.ts` 타입 재선언 | `governance.ts`에서 import |
| `window.confirm()` 승인 처리 | `<ApprovalGate>` 사용 |
| 코드 내 Secret 하드코딩 | `settings.*` 경유 |

### 9.6 Cursor AI 참조 지침

```

@phase3_context.json       ← 전체 구현 기준
```
